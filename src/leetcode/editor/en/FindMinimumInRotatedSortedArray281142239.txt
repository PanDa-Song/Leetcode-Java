class Solution {
    public int findMin(int[] nums) {
    if (nums.length == 0 || nums == null){
      return -1;
    }
    int end = nums.length - 1;
    int start = 0;
    while (start + 1 < end){
      int mid  = start + (end - start) / 2;
      if (nums[mid] < nums[end]){
        end = mid; 
      }else{
        start = mid;
      }
    }
    if (nums[start] < nums[end]){
      return nums[start];
    }
    return nums[end];
  }
}
